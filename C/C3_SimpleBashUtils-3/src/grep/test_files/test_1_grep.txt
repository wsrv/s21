
int s21_strlen(const char *str) {
    int size = 0;
    while (str[size] != '\0')
        size++;
    return size;
}
int s21_strcmp(const char *str1, const char *str2) {
    int cmp = 0;
    int s1 = s21_strlen(str1);
    int s2 = s21_strlen(str2);
    cmp = (s1 == s2 ? 0 : (s1 > s2 ? 1 : (-1)));
    return cmp;
}
